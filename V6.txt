aws_access_key = ""

aws_region = "us-east-1"

def get_cost_explorer_client():
    return boto3.client(
        'ce',
        aws_access_key_id=aws_access_key,
        aws_secret_access_key=aws_secret_key,
        region_name=aws_region
    )

class CostUsageDataPoint:
    def __init__(self, date: str, amount: str, unit: str):
        self.Date = date
        self.Amount = amount
        self.Unit = unit

class RootCauseEntry:
    def __init__(self, service, region, usage_type, linked_account, linked_account_name, cost_impact, tags=None):
        self.Service = service
        self.Region = region
        self.UsageType = usage_type
        self.LinkedAccount = linked_account
        self.LinkedAccountName = linked_account_name
        self.Tags = tags or {}  # Initialize with empty dict if no tags provided
        self.CostImpact = f"{float(cost_impact):.2f}" if cost_impact else "0.00"
        self.CostUsageGraph: List[CostUsageDataPoint] = []
        self.TotalCostFromGraph = "0.00"
        self.AverageDailyCostFromGraph = "0.00"

    def calculate_costs_from_graph(self):
        if not self.CostUsageGraph:
            return
        
        total_cost = sum(float(point.Amount) for point in self.CostUsageGraph)
        days = len(self.CostUsageGraph)
        
        self.TotalCostFromGraph = f"{total_cost:.2f}"
        self.AverageDailyCostFromGraph = f"{(total_cost / days):.2f}" if days > 0 else "0.00"

class AnomalyEntry:
    def __init__(self, anomaly_id, start_date, end_date, impact, duration, root_causes, expected_spend=None, actual_spend=None):
        self.AnomalyId = anomaly_id
        self.StartDate = start_date
        self.EndDate = end_date
        self.LastDetectedDate = end_date
        self.DurationInDays = duration
        self.TotalCostImpact = f"{impact:.2f}"
        self.ExpectedSpend = f"{expected_spend:.2f}" if expected_spend is not None else "0.00"
        self.ActualSpend = f"{actual_spend:.2f}" if actual_spend is not None else "0.00"
        self.AverageDailyCost = f"{(impact / duration):.2f}" if duration > 0 else f"{impact:.2f}"
        self.Currency = "USD"
        self.RootCauses = root_causes

async def fetch_anomalies():
    client = get_cost_explorer_client()
    output = []
    start = (datetime.now(UTC) - timedelta(days=90)).strftime('%Y-%m-%d')
    end = datetime.now(UTC).strftime('%Y-%m-%d')

    next_token = None

    while True:
        request = {
            'DateInterval': {'StartDate': start, 'EndDate': end}
        }
        if next_token:
            request['NextPageToken'] = next_token

        response = client.get_anomalies(**request)

        for anomaly in response.get('Anomalies', []):
            anomaly_start = datetime.strptime(anomaly['AnomalyStartDate'].split('T')[0], '%Y-%m-%d')
            anomaly_end = datetime.strptime(anomaly['AnomalyEndDate'].split('T')[0], '%Y-%m-%d')
            duration = (anomaly_end - anomaly_start).days + 1 or 1

            impact = float(anomaly.get('Impact', {}).get('TotalImpact', 0))
            expected_spend = float(anomaly.get('Impact', {}).get('ExpectedValue', 0))
            actual_spend = float(anomaly.get('Impact', {}).get('MaxImpact', 0))
            root_causes = []

            for rc in anomaly.get('RootCauses', []):
                # Get tags from the root cause data
                tags = {}
                if 'Tags' in rc:
                    tags = rc['Tags']
                elif 'ResourceTags' in rc:
                    tags = rc['ResourceTags']
                
                root_cause = RootCauseEntry(
                    service=rc.get('Service'),
                    region=rc.get('Region'),
                    usage_type=rc.get('UsageType'),
                    linked_account=rc.get('LinkedAccount'),
                    linked_account_name=rc.get('LinkedAccountName'),
                    cost_impact=rc.get('Impact', {}).get('Contribution', 0),
                    tags=tags
                )

                root_cause.CostUsageGraph = await fetch_cost_usage_for_root_cause(
                    client, anomaly_start, anomaly_end, root_cause
                )

                root_causes.append(root_cause)

            entry = AnomalyEntry(
                anomaly_id=anomaly.get('AnomalyId'),
                start_date=anomaly.get('AnomalyStartDate'),
                end_date=anomaly.get('AnomalyEndDate'),
                impact=impact,
                duration=duration,
                root_causes=root_causes,
                expected_spend=expected_spend,
                actual_spend=actual_spend
            )

            output.append(entry)

        next_token = response.get('NextPageToken')
        if not next_token:
            break

    return output

async def fetch_cost_usage_for_root_cause(client, start_date, end_date, root_cause):
    filter_list = []

    def add_filter(key, value):
        if value:
            filter_list.append({
                'Dimensions': {
                    'Key': key,
                    'Values': [value]
                }
            })

    add_filter('SERVICE', root_cause.Service)
    add_filter('REGION', root_cause.Region)
    add_filter('USAGE_TYPE', root_cause.UsageType)
    add_filter('LINKED_ACCOUNT', root_cause.LinkedAccount)

    if len(filter_list) == 1:
        filters = filter_list[0]
    elif len(filter_list) > 1:
        filters = {'And': filter_list}
    else:
        filters = None

    try:
        # Define the specific tags we want to fetch
        required_tags = {
            'SupportEmail': 'Support Email',
            'CostCenter': 'Cost Center',
            'Application': 'Application',
            'LOB': 'Line of Business',
            'Owner': 'Owner',
            'Product': 'Product',
            'Project': 'Project',
            'ProjectName': 'Project Name'
        }
        
        # Initialize tags dictionary with empty values for all required tags
        tags = {tag: "" for tag in required_tags.keys()}
        
        # First, get the cost data points (without tags)
        cost_response = client.get_cost_and_usage(
            TimePeriod={
                'Start': start_date.strftime('%Y-%m-%d'),
                'End': (end_date + timedelta(days=1)).strftime('%Y-%m-%d')
            },
            Granularity='DAILY',
            Metrics=['UnblendedCost'],
            Filter=filters
        )

        # Make separate API calls for each pair of tags
        tag_keys = list(required_tags.keys())
        for i in range(0, len(tag_keys), 2):
            # Only include up to two tags in each call
            current_tags = tag_keys[i:i+2]
            
            # Create tag filters for the current pair
            tag_filters = []
            for tag in current_tags:
                tag_filters.append({
                    'Tags': {
                        'Key': tag,
                        'MatchOptions': ['EXISTS']
                    }
                })
            
            # Combine tag filters with existing filters
            combined_filters = filters
            if tag_filters:
                if combined_filters:
                    combined_filters = {
                        'And': [
                            combined_filters,
                            {'Or': tag_filters}
                        ]
                    }
                else:
                    combined_filters = {'Or': tag_filters}

            try:
                # Get cost and usage data with current tag group
                response = client.get_cost_and_usage(
                    TimePeriod={
                        'Start': start_date.strftime('%Y-%m-%d'),
                        'End': (end_date + timedelta(days=1)).strftime('%Y-%m-%d')
                    },
                    Granularity='DAILY',
                    Metrics=['UnblendedCost'],
                    Filter=combined_filters,
                    GroupBy=[{'Type': 'TAG', 'Key': tag} for tag in current_tags]
                )
                
                # Extract tags from the response
                if 'ResultsByTime' in response and response['ResultsByTime']:
                    for result in response['ResultsByTime']:
                        if 'Groups' in result:
                            for group in result['Groups']:
                                for dimension in group.get('Keys', []):
                                    if dimension.startswith('TAG$'):
                                        # Format is TAG$Key$Value
                                        parts = dimension.split('$')
                                        if len(parts) == 3:
                                            tag_key = parts[1]
                                            tag_value = parts[2]
                                            if tag_value and tag_value.strip():  # Only add non-empty values
                                                tags[tag_key] = tag_value
            except Exception as e:
                print(f"Warning: Error fetching tags {current_tags} for {root_cause.Service}: {e}")
                continue

        # Update root cause tags
        root_cause.Tags = tags

        # Process cost data points
        points = []
        for time_period in cost_response['ResultsByTime']:
            amount_data = time_period['Total'].get('UnblendedCost', {})
            points.append(CostUsageDataPoint(
                date=time_period['TimePeriod']['Start'],
                amount=amount_data.get('Amount', "0"),
                unit=amount_data.get('Unit', "USD")
            ))
        return points

    except Exception as e:
        print(f"Error fetching cost usage for root cause {root_cause.Service}: {e}")
        return []

def save_to_json(entries, filename):
    def default_serializer(o):
        if isinstance(o, (AnomalyEntry, RootCauseEntry, CostUsageDataPoint)):
            return o.__dict__
        return str(o)

    with open(filename, 'w') as f:
        json.dump(entries, f, indent=4, default=default_serializer)

    print(f"Saved {len(entries)} entries to {filename}")

    if rc.TotalCostFromGraph == "0.00":
        rc.calculate_costs_from_graph()

    try:
        cost_impact = float(rc.CostImpact)
        formatted_cost_impact = f"${cost_impact:.2f}"
    except (ValueError, TypeError):
        formatted_cost_impact = "$0.00"
    
    # Format tags for display with specific styling
    tags_html = ""
    if rc.Tags:
        # Define tag display order and labels
        tag_display_order = [
            ('SupportEmail', 'Support Email'),
            ('CostCenter', 'Cost Center'),
            ('Application', 'Application'),
            ('LOB', 'Line of Business'),
            ('Owner', 'Owner'),
            ('Product', 'Product'),
            ('Project', 'Project'),
            ('ProjectName', 'Project Name')
        ]
        
        # Create tag display with specific styling
        for tag_key, tag_label in tag_display_order:
            tag_value = rc.Tags.get(tag_key, '')
            if tag_value and tag_value.strip():
                tags_html += f"""
                    <div class="tag">
                        <span class="tag-key">{tag_label}:</span>
                        <span class="tag-value">{tag_value}</span>

                    <div class="tag tag-empty">

                        <span class="tag-value">Not Set</span>

                """

        tags_html = """
            <div class="tag tag-empty">
                <span class="tag-key">No tags available</span>
            </div>
        """
    
    return f"""

            // Function to compress image to stay under 50KB
            async function compressImage(dataUrl) {
                return new Promise((resolve, reject) => {
                    const img = new Image();
                    img.onload = function() {
                        const canvas = document.createElement('canvas');
                        let width = img.width;
                        let height = img.height;
                        
                        // Start with a reasonable size
                        const maxWidth = 800;
                        if (width > maxWidth) {
                            height = Math.round((height * maxWidth) / width);
                            width = maxWidth;
                        }
                        
                        canvas.width = width;
                        canvas.height = height;
                        
                        const ctx = canvas.getContext('2d');
                        ctx.fillStyle = '#ffffff';
                        ctx.fillRect(0, 0, width, height);
                        
                        // Enable image smoothing for better quality
                        ctx.imageSmoothingEnabled = true;
                        ctx.imageSmoothingQuality = 'high';
                        
                        ctx.drawImage(img, 0, 0, width, height);
                        
                        // Try different quality levels until we get under 50KB
                        const qualities = [0.7, 0.5, 0.3, 0.2];
                        let currentQuality = 0;
                        
                        function tryCompress() {
                            if (currentQuality >= qualities.length) {
                                // If we've tried all qualities and still too large, reduce dimensions
                                const finalCanvas = document.createElement('canvas');
                                finalCanvas.width = Math.round(width * 0.7);
                                finalCanvas.height = Math.round(height * 0.7);
                                const finalCtx = finalCanvas.getContext('2d');
                                finalCtx.fillStyle = '#ffffff';
                                finalCtx.fillRect(0, 0, finalCanvas.width, finalCanvas.height);
                                finalCtx.drawImage(canvas, 0, 0, finalCanvas.width, finalCanvas.height);
                                resolve(finalCanvas.toDataURL('image/jpeg', 0.2));
                                return;
                            }
                            
                            const quality = qualities[currentQuality];
                            const compressedDataUrl = canvas.toDataURL('image/jpeg', quality);
                            const base64str = compressedDataUrl.split(',')[1];
                            const decodedSize = atob(base64str).length;
                            
                            if (decodedSize <= 45000) { // Leave some margin for other variables
                                resolve(compressedDataUrl);
                            } else {
                                currentQuality++;
                                tryCompress();
                            }
                        }
                        
                        tryCompress();
                    };
                    img.onerror = reject;
                    img.src = dataUrl;
                });
            }

                try {
                    const searchText = (document.getElementById('searchText')?.value || '').toLowerCase().trim();

                    const minCost = parseFloat(document.getElementById('minCost')?.value) || 0;

                        const costText = header.querySelectorAll('.header-value')[2]?.textContent?.trim() || '';

                        const [startDateStr, endDateStr] = dateText.split(' to ').map(d => d.trim());
                        const anomalyStart = new Date(startDateStr);
                        const anomalyEnd = new Date(endDateStr);
                        let filterStart = null, filterEnd = null;
                        if (dateRange && dateRange.includes(' to ')) {
                            const [filterStartStr, filterEndStr] = dateRange.split(' to ').map(d => d.trim());
                            if (filterStartStr && filterEndStr) {
                                filterStart = new Date(filterStartStr);
                                filterEnd = new Date(filterEndStr);
                                if (filterEnd) filterEnd.setHours(23, 59, 59, 999);
                            }
                        }
                        const services = Array.from(content.querySelectorAll('.root-cause'))
                            .map(rc => rc.querySelector('.root-cause-title')?.textContent?.trim() || '')

                        const matchesSearch = !searchText || 
                            anomalyId.toLowerCase().includes(searchText) ||
                            dateText.toLowerCase().includes(searchText) ||
                            costText.toLowerCase().includes(searchText);

            start_readable = start_dt.strftime('%m/%d/%Y')
            end_readable = end_dt.strftime('%m/%d/%Y')
        except Exception as e:
            start_readable = entry.StartDate.split('T')[0]
            end_readable = entry.EndDate.split('T')[0]

        html += generate_accordion_html(entry, idx, start_readable, end_readable)

            chart_id = f"chart_{idx}_{rc_idx}"
            chart_idx += 1
            
            # Prepare data for Chart.js
            labels = [point.Date for point in rc.CostUsageGraph]

            """

            }})();
            </script>
